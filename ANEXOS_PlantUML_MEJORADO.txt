package "📦 Modelos (src/app/models)" {

  class Producto {
    - id: number
    - nombre: string
    - descripcion: string
    - precio: number
    - categoria: string
    - imagen: string
    - stock: number
  }

  class CarritoItem {
    - producto: Producto
    - cantidad: number
    ---
    + total(): number
  }

  class PurchaseItem {
    - producto_id: number
    - nombre_producto: string
    - categoria: string
    - cantidad: number
    - precio_unitario: number
    - importe: number
  }

  class PurchaseOrder {
    - id: number
    - folio: string
    - paypal_orden_id: string
    - paypal_estado: string
    - subtotal: number
    - iva: number [IVA 16%]
    - total: number
    - fecha: Date
    - items: PurchaseItem[]
  }

  class PayPalPayload {
    - intent: string = "CAPTURE"
    - purchase_units: PurchaseUnit[]
  }

  class PurchaseUnit {
    - amount: Amount
    - items: PayPalItem[]
  }

  class Amount {
    - currency_code: string = "MXN"
    - value: string
  }

  class PayPalItem {
    - name: string
    - quantity: string
    - unit_amount: UnitAmount
  }

  class UnitAmount {
    - currency_code: string = "MXN"
    - value: string
  }
}

title Diagrama Entidad-Relación — Base de Datos MariaDB\n(Tablas: productos, pedidos, detalles_pedido)

skinparam entity {
  BackgroundColor #EEF4FF
  BorderColor #3A6BC9
  FontColor #1A1A2E
  ArrowColor #3A6BC9
}

skinparam backgroundColor #FAFAFA

entity "productos" as PROD {
  * id : INT(11) [PK]
  --
  nombre : VARCHAR(255) [NOT NULL]
  descripcion : TEXT
  precio : DECIMAL(10,2) [NOT NULL]
  categoria : VARCHAR(100)
  imagen : VARCHAR(500)
  stock : INT(11) [DEFAULT 0]
  created_at : DATETIME [DEFAULT CURRENT_TIMESTAMP]
  updated_at : DATETIME [ON UPDATE CURRENT_TIMESTAMP]
}

entity "pedidos" as PED {
  * id : INT(11) [PK]
  --
  folio : VARCHAR(30) [UNIQUE, NOT NULL]
  cliente_nombre : VARCHAR(255)
  cliente_email : VARCHAR(255)
  paypal_orden_id : VARCHAR(100) [UNIQUE]
  paypal_estado : VARCHAR(50) [ENUM: CREATED, APPROVED, COMPLETED, FAILED]
  subtotal : DECIMAL(10,2) [NOT NULL]
  iva : DECIMAL(10,2) [16% calulado automáticamente]
  total : DECIMAL(10,2) [NOT NULL]
  fecha : DATETIME [DEFAULT CURRENT_TIMESTAMP]
  notas : TEXT
}

entity "detalles_pedido" as DET {
  * id : INT(11) [PK]
  --
  # pedido_id : INT(11) [FK → pedidos.id]
  # producto_id : INT(11) [FK → productos.id]
  nombre_producto : VARCHAR(255) [snapshot at order time]
  categoria : VARCHAR(100)
  cantidad : INT(11) [NOT NULL]
  precio_unitario : DECIMAL(10,2) [snapshot at order time]
  importe : DECIMAL(10,2) [cantidad * precio_unitario]
}

' Relaciones
PED "1" ||--o{ DET : "1:N\ntiene\ndetalles"
PROD "1" ||--o{ DET : "1:N\nreferenciado\nen"

note right of PROD
  Catálogo principal de productos.
  Foto almacenada como URL en 'imagen'.
  Stock actualizable por administrador.
end note

note right of PED
  Registro de cada transacción/orden.
  Folio único formato: ORD-{timestamp}.
  paypal_estado sincronizado con PayPal API.
end note

note right of DET
  Líneas de detalle de pedido.
  Guarda snapshot de nombre y precio
  en momento de la compra (histórico).
end note

@startuml AnexoC2_FlujoCarrito
title Flujo — Módulo Carrito con PayPal SDK Dinámico

start
  :Usuario hace click en ícono de carrito;
  :Navega a /carrito;

  :CarritoComponent.ngOnInit();
  :Observa CarritoService.groupedItems;

  if (¿Carrito vacío?) then (Sí)
    :Muestra "Carrito vacío";
    :Botón "Ir a catálogo";
    stop
  else (No — hay items)
    :Itera sobre groupedItems;
    :Renderiza cada item con cantidad;
    :Muestra subtotal = SUM(precio × cantidad);
    :Calcula impuestos = subtotal × 0.16;
    :Muestra total = subtotal + impuestos;
  endif

  :Usuario scroll abajo;

  :ngAfterViewInit() se ejecuta;
  :loadPayPalSDK() → carga script dinámicamente;
  note right
    <script src="https://www.paypal.com/sdk/js?...">
    Inyecta en <head> si no existe
  end note

  :setInterval(200ms) monitorea cambios en carrito;

  if (¿window.paypal disponible?) then (Sí)
    :paypal.Buttons({...}) crea botones;
    :createOrder: POST /api/paypal/create-order;
    :onApprove: captura el pago;
    :Renderiza botón en #paypal-buttons-container;
  else (No)
    :Reintenta en siguiente ciclo;
  endif

  :Usuario ve botón "Pagar con PayPal";
  :Hace click;

  :createOrder callback:;
  :POST /api/paypal/create-order;
  :Body: {intent: "CAPTURE", purchase_units: [...] };

  :PayPal retorna {id: orderID};
  :Usuario aprobó en popup PayPal;

  :onApprove callback:;
  :POST /api/paypal/capture-order {orderID};

  if (¿Captura OK?) then (Sí)
    :PayPal retorna detalles de transacción;
    :POST /api/paypal/guardar-pedido;
    :Backend inserta en BD: pedidos + detalles_pedido;

    :CarritoComponent.generarXMLRecibo();
    :Crea <recibo> XML con estructura jerárquica;

    :HistorialComprasService.guardarCompra(PurchaseOrder);
    :Actualiza Signal compras[];

    :CarritoService.limpiarCarrito();
    :productosSignal = [];
    :itemCount = 0;

    :Muestra toast de éxito;
    :Redirige a /historial (auto);

  else (No/Error)
    :onError callback;
    :Muestra mensaje "Pago no capturado";
    :Usuario puede reintentar;
  endif

stop
@enduml

  RAÍZ del documento XML
  Generado por: CarritoComponent.generarXMLRecibo()

class "<encabezado>" as ENC {
  --
  <folio> : string
    ej: ORD-1777507322236
  <fecha> : dateTime (ISO 8601)
    ej: 2026-05-11T14:30:00-06:00
  <estado> : string
    COMPLETED | PENDING | FAILED
  <paypal_id> : string
    ID de orden desde PayPal
}

class "<emisor>" as EMI {
  --
  <nombre> : string = "La Casa del Perfume"
  <sistema> : string = "Angular 21 + Node.js"
  <moneda> : string = "MXN"
  <rfc> : string (opcional)
  <domicilio> : string (opcional)
}

class "<productos>" as PRODS {
  Contenedor de items
  Puede tener 1 a N elementos <producto>
}

class "<producto>" as PROD {
  --
  <producto_id> : number
  <nombre> : string
  <categoria> : string
  <cantidad> : integer ≥ 1
  <precio_unitario> : decimal
    (precisión: 2 decimales)
  <importe> : decimal
    = cantidad × precio_unitario
}

class "<totales>" as TOT {
  --
  <subtotal> : decimal
    SUM(importe de cada producto)
  <iva tasa="16%"> : decimal
    = subtotal × 0.16
  <total> : decimal
    = subtotal + iva
}

class "<pie>" as PIE {
  --
  <generado_por> : string = "CarritoComponent"
  <timestamp> : datetime (ISO 8601)
  <nota> : string
    "Este es un recibo digital..."
}
